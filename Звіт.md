# 2. Вилучення простої випадкової вибірки без повернення

sample_variance = np.var(sample_precip, ddof=1)
error_variance = abs(pop_variance - sample_variance)

# Виведення результатів

np.random.seed(7)

# 1. Генерація вибірки з біноміального розподілу
# n = 30 днів у місяці, p = 0.3 ймовірність дощу в окремий день
rainy_days = np.random.binomial(n=30, p=0.3, size=2000)

# 2. Знаходження унікальних значень та частот
unique_vals, counts = np.unique(rainy_days, return_counts=True)

print("\nВідносні частоти:")
print(np.round(rel_freqs, 3))

# 3. Побудова ECDF
variation_series_days = np.sort(rainy_days)

def evaluate_ecdf(x_val, data):
    return np.sum(data <= x_val) / len(data)

ecdf_val = evaluate_ecdf(x_target, variation_series_days)

print(f"\nЗначення ECDF у точці {x_target}: {ecdf_val:.3f}")

Описові статистики та агрегація даних у PandasПостановка задачіНеобхідно обробити дані з метеостанцій різних регіонів та проаналізувати температурні режими. Для цього створюється DataFrame із категоріальною колонкою (Регіон), колонкою зі штучними пропусками (Температура_С) та колонкою Вологість_%. Потрібно заповнити пропуски середнім арифметичним, відфільтрувати дні з вологістю понад 60%, згрупувати за регіоном та знайти середню температуру.Покрокове виконання та програмний кодPythonimport pandas as pd
import numpy as np

np.random.seed(7)

# 1. Створення DataFrame зі згенерованими даними
data = {
    'Регіон': np.random.choice(['Захід', 'Центр', 'Південь'], size=20),
    'Температура_С': np.random.choice([12.5, 15.0, 18.2, 22.5, 25.1, np.nan], size=20, p=[0.15, 0.15, 0.2, 0.2, 0.1, 0.2]),
    'Вологість_%': np.random.randint(40, 95, size=20)
}
df = pd.DataFrame(data)

# 2. Обробка пропусків (Imputation) середнім значенням
mean_temp = df['Температура_С'].mean()
df['Температура_С'] = df['Температура_С'].fillna(mean_temp)

print(f"Середня температура для заповнення пропусків: {mean_temp:.2f} °C\n")

# 3. Фільтрація: дні з вологістю понад 60%
df_filtered = df[df['Вологість_%'] > 60]

# 4. Групування за регіоном і знаходження середньої температури
grouped_stats = df_filtered.groupby('Регіон')['Температура_С'].mean().round(2).reset_index()
grouped_stats.columns = ['Регіон', 'Середня температура (°C)']

print("Результати групування (дні з вологістю > 60%):")
print(grouped_stats.to_string(index=False))
